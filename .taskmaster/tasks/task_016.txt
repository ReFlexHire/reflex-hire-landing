// 온보딩 크레딧 계산
const calculateOnboardingCredit = (salary) => {
  const fee = salary * 0.1; // 10% 수수료
  const creditRatio = 0.3; // 수수료의 30%를 크레딧으로
  return fee * creditRatio;
};

// 패키지 자동 매칭
const getPackageForSalary = (salary) => {
  if (salary >= 100000000) return 'premium';
  if (salary >= 50000000) return 'standard';
  return 'basic';
};
