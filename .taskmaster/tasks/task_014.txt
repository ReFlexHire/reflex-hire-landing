app.get('/admin/dashboard', checkRole(['ADMIN']), async (req, res) => {
  const stats = await db.query(`
    SELECT 
      COUNT(*) as total_jobs,
      SUM(CASE WHEN status = 'completed' THEN 1 ELSE 0 END) as completed_jobs
    FROM jobs
  `);
  res.json(stats.rows[0]);
});
