```javascript
// 벤더 주문 API
const createVendorOrder = async (packageId, deliveryInfo) => {
  const vendor = await getVendorByPackage(packageId);
  const orderData = {
    items: packageId,
    delivery_address: deliveryInfo,
    callback_url: `${process.env.API_URL}/webhook/vendor-status`
  };
  
  return await axios.post(vendor.api_endpoint, orderData, {
    headers: { 'Authorization': `Bearer ${vendor.api_key}` }
  });
};
```
