```javascript
// 권한 체크 미들웨어
const checkRole = (roles) => {
  return (req, res, next) => {
    if (!roles.includes(req.user.role)) {
      return res.status(403).json({ error: '접근 권한이 없습니다' });
    }
    next();
  };
};
```
